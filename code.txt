
# Prompt
function prompt() {
	local red="\[\e[1;31m\]"
	local green="\[\e[1;32m\]"
	local cyan="\[\e[1;36m\]"
	local reset="\[\e[m\]"
	if [[ -z $VIRTUAL_ENV ]]
	then
		local venv=""
	else
		local venv_name=$(echo $VIRTUAL_ENV | awk -F "/" '{print $NF}')
		local venv="$green($venv_name)$reset "
	fi
	if [[ $? -eq 0 ]]
	then
		local error_color=$green
	else
		local error_color=$red
	fi
	PS1=" $venv$cyan\w$reset\n $error_color$reset "
}
PROMPT_COMMAND=prompt
